## 내가 푼 코드

```python
def solution(dartResult):
    num = [0]
    bonus = [0]
    option = [' ']
    
    result = 0
    
    flag = True
    
    for i in range(len(dartResult)):
        if flag == False:
            flag = True
            continue
        
        try:
            if dartResult[i].isdecimal():
                num.append(int(dartResult[i]))
                if dartResult[i+1].isdecimal():
                    flag = False
                    num[-1] = 10
            elif dartResult[i] in ['S','D','T']:
                if dartResult[i] == 'S':
                    bonus.append(1)
                elif dartResult[i] == 'D':
                    bonus.append(2)
                elif dartResult[i] == 'T':
                    bonus.append(3)
                    
                if dartResult[i+1] not in ['*', '#']:
                    option.append(' ')
            else:
                option.append(dartResult[i])      
        except:
            continue
        
    result = [0]
    
    for i in range(1,4):
        result.append(num[i]**bonus[i])
        
    if len(option) == 3:
        option.append(' ')
    
    for i in range(1,4):
        if option[i] == '*':
            result[i] *= 2
            result[i-1] *= 2
        elif option[i] == '#':
            result[i] *= -1
    
    return sum(result)
```

## 타인이 푼 코드

### 타인이 푼 방법 1 - 정규 표현식을 이용한 풀이

```python
import re

def solution(dartResult):
    bonus = {'S' : 1, 'D' : 2, 'T' : 3}
    option = {'' : 1, '*' : 2, '#' : -1}
    p = re.compile('(\d+)([SDT])([*#]?)')
    dart = p.findall(dartResult)
    for i in range(len(dart)):
        if dart[i][2] == '*' and i > 0:
            dart[i-1] *= 2
        dart[i] = int(dart[i][0]) ** bonus[dart[i][1]] * option[dart[i][2]]

    answer = sum(dart)
    return answer
```

정규표현식을 통해 더 쉽게 계산하게됨

### 타인이 푼 방법 2 - 10을 변환 후 더 간단히 계산

```python
def solution(dartResult):
    point = []
    answer = []
    dartResult = dartResult.replace('10','k')
    point = ['10' if i == 'k' else i for i in dartResult]
    print(point)

    i = -1
    sdt = ['S', 'D', 'T']
    for j in point:
        if j in sdt :
            answer[i] = answer[i] ** (sdt.index(j)+1)
        elif j == '*':
            answer[i] = answer[i] * 2
            if i != 0 :
                answer[i - 1] = answer[i - 1] * 2
        elif j == '#':
            answer[i] = answer[i] * (-1)

            answer.append(int(j))
            i += 1
    return sum(answer)
```
